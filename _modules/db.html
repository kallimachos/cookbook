[docs]def createdb(database):
    """Create an sqlite database."""
    con = sql.connect(database)
    with con:
        cur = con.cursor()
        cur.execute("""DROP TABLE IF EXISTS Users""")
        cur.execute("""CREATE TABLE Users
                    (id INTEGER PRIMARY KEY,
                    name TEXT NOT NULL,
                    phone INT,
                    email TEXT unique,
                    image BLOB)""")
    return(True)
